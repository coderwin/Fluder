  • Vue Component 
import todoStore from 'stores/todoStore';
//...
methods:{
  //add时调用
  addTodo(e){
    todoAction.addTodo({
      text: e.target.value,
      done: false
    });
  },
  _onTodoChange(){
    this.items = todoStore.getAll()
  }
},
ready (){
  todoStore.addChangeListener(this._onTodoChange)
}
